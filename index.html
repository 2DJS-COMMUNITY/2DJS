Player

const player = new 2DJS.Player(100, 100);
player.rigid(true);
player.setControls({ left:"a", right:"d", up:"w" });

UI

const button = new 2DJS.UI.Create("button", {
  text: "Play",
  x: 100, y: 50, width: 120, height: 40
});
scene.add(button);
